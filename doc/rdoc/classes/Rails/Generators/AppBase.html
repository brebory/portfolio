     # File railties/lib/rails/generators/app_base.rb, line 246
246:       def bundle_command(command)
247:         say_status :run, "bundle #{command}"
248: 
249:         # We are going to shell out rather than invoking Bundler::CLI.new(command)
250:         # because `rails new` loads the Thor gem and on the other hand bundler uses
251:         # its own vendored Thor, which could be a different version. Running both
252:         # things in the same process is a recipe for a night with paracetamol.
253:         #
254:         # We use backticks and #print here instead of vanilla #system because it
255:         # is easier to silence stdout in the existing test suite this way. The
256:         # end-user gets the bundler commands called anyway, so no big deal.
257:         #
258:         # Thanks to James Tucker for the Gem tricks involved in this call.
259:         print `"#{Gem.ruby}" -rubygems "#{Gem.bin_path('bundler', 'bundle')}" #{command}`
260:       end
